What is Express.js?

    A fast, minimal web framework for Node.js.
    Provides middleware, routing, error handling, etc.

    const express = require("express");
    const app = express();

    // Built-in middleware to parse JSON
    app.use(express.json());

    // 1. Application-level middleware
    app.use((req, res, next) => {
        console.log(`ðŸ“Œ ${req.method} ${req.url} at ${new Date().toISOString()}`);
        next(); // pass control to next middleware/route
    });

    // 2. Router-level middleware
    const router = express.Router();

    router.use((req, res, next) => {
        console.log("ðŸ”¹ Router-level middleware executed");
        next();
    });

    router.get("/hello", (req, res) => {
        res.send("Hello from Router with middleware!");
    });

    app.use("/api", router);

    // 3. Route with multiple middleware
    function checkAuth(req, res, next) {
        const token = req.headers["authorization"];
        if (token === "12345") {
            next(); // user authenticated and pass control to route handler
        } else {
            res.status(401).json({ error: "Unauthorized" });
        }
    }

    app.get("/secure", checkAuth, (req, res) => {
        res.send("âœ… You have access to secure route!");
    });

    // 4. Error-handling middleware
    app.use((err, req, res, next) => {
        console.error("âŒ Error:", err.message);
        res.status(500).json({ error: "Something went wrong!" });
    });

    // Start server
    app.listen(3000, () => {
        console.log("ðŸš€ Server running at http://localhost:3000");
    });
